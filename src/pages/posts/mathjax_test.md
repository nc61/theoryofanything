$$
\begin{align}
    \nabla^2 \mathbf{A} - \frac{1}{c^2}\frac{\partial^2 \mathbf{A}}{\partial t^2} &= \mu_0 \mathbf{J}(\mathbf{r}) \\
    \nabla^2 \varphi - \frac{1}{c^2}\frac{\partial^2 \varphi}{\partial t^2} &= \frac{1}{\epsilon_0}\rho(\mathbf{r}),
\end{align}
$$

we find on the left-hand side the same linear differential operator acting on each component of the electromagnetic four-potential $(\mathbf{A},\varphi)$. Denoting this operator $\hat{L}(\mathbf{r},t) := \nabla^2 - (1/c^2)\partial_t^2$, it is easy to generalize the information here to apply to an arbitrary differential operator. With this definition, \cref{eq:emg} becomes

$$
\begin{equation}\label{eq:gf_deL}
    \hat{L}(\mathbf{r},t)f(\mathbf{r},t) = s(\mathbf{r},t),
\end{equation}
$$

where $f(\mathbf{r},t)$ is a field component $s(\mathbf{r},t)$ is the corresponding source.

We can now construct the Green's function for the wave equation in \cref{eq:gf_deL}, which is just the field generated by an impulse at position $\mathbf{r}'$ and time $t$. The Green's function is defined with the notation $G(\mathbf{r},t;\mathbf{r}',t')$, representing the field at position $\mathbf{r}$ and $t$ due to a point source at $\mathbf{r}'$ and $t'$. Directly apply \cref{eq:gf_deL} to find the equation

$$
\begin{equation}\label{eq:gf_L}
    \hat{L}(\mathbf{r},t)G(\mathbf{r},t;\mathbf{r}',t') = \delta(\mathbf{r} - \mathbf{r}')\delta(t - t').
\end{equation}
$$

$$
\begin{equation}\label{eq:gf_delta}
    \rho(\mathbf{r},t) = \int \rho(\mathbf{r}',t')\delta(\mathbf{r} - \mathbf{r}')\delta(t - t') d\mathbf{r}' dt'.
\end{equation}
$$

This integral is trivially true, but it is convenient because it allows us to act on each side of \cref{eq:gf_L} by the integral operator $\int d\mathbf{r}' dt' \rho(\mathbf{r}',t')$ for some arbitrary charge distribution $\rho$. The end result is 

$$
\begin{equation}\label{eq:gf_LG}
    \hat{L}(\mathbf{r}',t')\int G(\mathbf{r},\mathbf{t};\mathbf{r}',t') \rho(\mathbf{r}',t') d\mathbf{r}'dt' = \rho(\mathbf{r},t).
\end{equation}
$$

The right hand side of \cref{eq:gf_LG} came from \cref{eq:gf_delta} and the left hand side was simplifed by noting that the linear operator acts on unprimed coordinates so it can be pulled out of the integral. Finally comparing \cref{eq:gf_LG} to \cref{eq:gf_deL}, we find that we are able to find the field 

$$
\begin{equation}
    \varphi(\mathbf{r},t) = \int G(\mathbf{r},\mathbf{t};\mathbf{r}',t') \rho(\mathbf{r}',t') d\mathbf{r}'dt'
\end{equation}
$$

using the impulse response (Green's) function $G$.

Now we just specialize to the case of electromagnetic waves. Things are easier if we choose coordinates where $\mathbf{r}' = 0$, so we are looking to solve the equation
$$
\begin{equation}
    \left(\nabla^2 - \frac{1}{c^2}\frac{\partial^2}{\partial t^2}\right)G(\mathbf{r},t;\mathbf{r}',t') = \frac{1}{\epsilon_0}\delta(\mathbf{r} - \mathbf{r}')\delta(t - t')
\end{equation}
$$
We solve with a plane wave expansion of $G$
$$
\begin{equation}\label{eq:pwe}
    G(\mathbf{r},t;\mathbf{r}',t') = \frac{1}{(2\pi)^4}\int \tilde{G}(\mathbf{k},\omega; \mathbf{r}',t') e^{i(\mathbf{k}\cdot \mathbf{r} - \omega t)}d\mathbf{k} d\omega,
\end{equation}
$$
where $\tilde{G}(\mathbf{k},\omega;\mathbf{r}',t')$ is the $3 + 1$-dimensional Fourier transform defined as
$$
\begin{equation}
    \tilde{G}(\mathbf{k},\omega;\mathbf{r}',t') = \int G(\mathbf{r},t; \mathbf{r}',t') e^{-i(\mathbf{k}\cdot \mathbf{r} - \omega t)}d\mathbf{k} d\omega.
\end{equation}
$$
Writing $G(\mathbf{r},t)$ using the inverse transform, we have the equation
$$
\begin{equation}
    \left(\nabla^2 - \frac{1}{c^2}\frac{\partial^2}{\partial t^2}\right)\int \tilde{G}(\mathbf{k},\omega; \mathbf{r}',t') e^{i(\mathbf{k}\cdot \mathbf{r} - \omega t)}d\mathbf{k}d\omega = \frac{1}{\epsilon_0}\int e^{-i(\mathbf{k}\cdot \mathbf{r}' - \omega t')} e^{i(\mathbf{k}\cdot \mathbf{r} - \omega t)} d\mathbf{k}d\omega.
\end{equation}
$$
In the transform representation, the operator acts on the plane wave component so that
$$
\begin{equation}
    \int \left[\left(\frac{\omega^2}{c^2} - k^2\right) \tilde{G}(\mathbf{k},\omega; \mathbf{r}',t') - \frac{e^{-i(\mathbf{k}\cdot \mathbf{r}' - \omega t')}}{\epsilon_0} \right]e^{i(\mathbf{k}\cdot \mathbf{r} - \omega t)}d\mathbf{k}d\omega = 0.
\end{equation}
$$
It follows from the orthonormality of the plane waves that the term in brackets must equal zero, which can then be rearranged for
$$
\begin{equation}\label{eq:gf_fr}
    \tilde{G}(\mathbf{k},\omega;\mathbf{r}',t') = \frac{c^2}{\epsilon_0}\frac{e^{-i(\mathbf{k}\cdot \mathbf{r}' - \omega t')}}{\omega^2 - c^2k^2}
\end{equation}
$$
Finally, use the inverse Fourier transform relation in \cref{eq:pwe} to compute $G(\mathbf{r},t;\mathbf{r}',t')$ from the result in \cref{eq:gf_fr}:
$$
\begin{equation}
    G(\mathbf{r},t;\mathbf{r}',t') = \frac{1}{(2\pi)^4}\frac{c^2}{\varepsilon_0}\int \frac{1}{\omega^2 - c^2 k^2} e^{i(\mathbf{k}\cdot(\mathbf{r} - \mathbf{r}') - \omega(t - t'))}d\mathbf{k} d\omega,
\end{equation}
$$
which is just the shifted version of $\mathscr{F}^{-1}\left\{G(\mathbf{k},\omega;0,0)\right\}(\mathbf{r} - \mathbf{r}', t-t')$. Consequently, it suffices to calculate the response $G(\mathbf{r},t;0,0)$ and replace $\mathbf{r}$ and $t$ by $\mathbf{r} - \mathbf{r}'$ and $t - t'$, respectively, in the final result. Compute this quantity by performing the integration using spherical coordinates:
$$
\begin{equation}\label{eq:G0}
    G(\mathbf{r},t;0,0) = \frac{1}{(2\pi)^4}\frac{c^2}{\epsilon_0} \int_{0}^\infty dk\int_{-\infty}^{\infty}\frac{k^2}{\omega^2 - c^2 k^2} e^{-i\omega t} d\omega \int_0^\pi e^{ikr\cos\theta} \sin \theta d\theta \int_{0}^{2\pi} d\phi.
\end{equation}
$$
The integral over $\phi$ in \cref{eq:G0} is trivial because the integrand does not depend on the azimuthal angle $\phi$. Integration over the polar angle $\theta$ is done by substituting $u = ikr \cos \theta$ and $du = -ikr \sin \theta d\theta$ $(\sin\theta d\theta = - du/ikr)$. What remains is the integral over $k$ and $\omega$,
$$
\begin{equation}\label{eq:G1}
    G(\mathbf{r},t;0,0) = \frac{2}{(2\pi)^3}\frac{c^2}{\epsilon_0}\frac{1}{ir} \int_{0}^\infty k \left(e^{ikr} - e^{-ikr}\right) dk\int_{-\infty}^{\infty}\frac{e^{-i\omega t}}{\omega^2 - c^2k^2}  d\omega.
\end{equation}
$$

We will first evaluate the rightmost integral in \cref{eq:G1} over $\omega$. For convenience, denote this quantity as $\Omega(k,t)$. To begin, factor the denominator as $\omega^2 - c^2k^2 = (\omega - ck)(\omega + ck)$. Because we have poles in the integrand, this integral must be evaluated using the principal value,
$$
\begin{equation}
    \Omega(k,t) = P.V.\int_{-\infty}^{\infty} \frac{e^{-i\omega t}}{(\omega - ck)(\omega + ck)}d\omega,
\end{equation}
$$
which removes a small symmetric region in the neighborhood of each singularity by altering the limits to
$$
\begin{equation}\label{eq:rlim}
	\Omega(k,t) = \lim_{R\rightarrow \infty} \lim_{\epsilon \rightarrow 0}\left(\int_{-R}^{-ck - \epsilon}d\omega + \int_{-ck + \epsilon}^{ck - \epsilon}d\omega + \int_{ck + \epsilon}^{R} d\omega \right) \frac{e^{-i\omega t}}{(\omega - ck)(\omega + ck)}.
\end{equation}
$$
Though the integral we want to evaluate is over the real line only, we can calculate it by integrating over a closed contour in the complex plane. To do so, replace the real frequency $\omega$ with the complex quantity $z$ so that the integrand becomes
$$
\begin{align}
	f(z) = \frac{e^{-iz t}}{(z - ck)(z + ck)}dz.
\end{align}
$$
We integrate over a closed closed contour $\mathscr{C}$ that contains the limits of the principal value integration along the real line by defining
$$
\begin{align}\label{eq:path}
	&\oint_{\mathcal{C}} f(z) dz = \Omega(k,t) + \lim_{R\rightarrow \infty} \lim_{\epsilon \rightarrow 0} \left(\int_{S(-ck,\epsilon)} dz + \int_{S(ck,\epsilon)} dz + \int_{S(0,R)} dz\right)f(z).
\end{align}
$$
Though the ordering is swapped, the path traverses the real line from $-\infty$ to $-ck -\varepsilon$, a small semicircle about the pole at $-ck$ ($S(-ck,\varepsilon)$), a path along the real line from $-ck + \varepsilon$ to $ck - \varepsilon$, a small semicircle about the pole at $ck$ ($S(ck,\varepsilon)$), the real line from $ck + \varepsilon$ to $\infty$, and a semicircular arc with infinite radius connecting from $+\infty$ to $-\infty$ ($S(R,\varepsilon)$).

If we define the excursions around each singularity so that $f(z)$ contains no poles within its boundaries, then $f(z)$ is analytic everywhere within the contour so the integral $\oint_{\mathcal{C}} f(z) dz = 0$ by the residue theorem. \Cref{eq:path} then rearranges to
$$
\begin{equation}
	\Omega(k,t) = - \lim_{\epsilon \rightarrow 0} \left(\int_{S(-ck,\epsilon)} dz + \int_{S(ck,\epsilon)} dz + \int_{S(0,R)} dz\right)f(z).
\end{equation}
$$
That is, we can calculate the desired integral $\Omega(k,t)$ by evaluating only the semicircular contours about each pole along with the infinite radius semicircle that connects the positive and negative real axes. Note that choosing contours that exclude poles from the integration region is the most convenient choice, but it is not required. We would find the same end result for $\Omega(k,t)$ if we included the poles inside the contour and accounted for the fact that the residue theorem must be applied with poles inside the integration region.

Next, we need to choose whether to close the contour in the upper or lower half plane. We are constrained by the requirement that the integral over the infinite semicircular is well defined, and that depends on the sign of the parameter $t$.

The constraints on the integration path arise from integration along the arc of radius $R$ over an angle $\pi$. To determine whether this arc should traverse through the upper- or lower-half plane depend on the asymptotic behavior of the numerator $f(z) = e^{-izt}$. Parametrizing in polar coordinates, the function along the arc is
$$
\begin{equation}
	f(z) = e^{-iR(\cos \theta + i \sin \theta)t} =  e^{-iR\cos \theta t}  e^{R \sin \theta t}.
\end{equation}
$$
The condition $t > 0$ requires the value of $\sin\theta$ to be negative, since positive values of $\sin\theta$ would cause the term $e^{R \sin \theta t}$ to diverge as $R \rightarrow \infty$. Consequently, we must choose a contour that closes in the lower half plane for which $e^{R \sin \theta t} \rightarrow 0$ as $R\rightarrow \infty$. 
$$
\begin{equation}
	\left|\int_{S(0,R)} e^{- i R t \cos \theta} e^{-R t \sin \theta} dz \right| \leq \int \left| R e^{-R t}\right||d\theta| \leq \pi \left|R e^{-Rt}\right|
\end{equation}
$$
Taking the limit as $R\rightarrow \infty$, we find that the exponential dominates so that $\lim_{R\rightarrow \infty} |Re^{-Rt}| = 0$. As a result, we must close the contour by a semicircle in the lower half plane. In this case, the semicircular paths around the poles must also extend into the lower half plane so that the contour contains no singularities. Around the pole at $-ck$, we define $z' = z + ck$ so that $dz' = dz$ and
$$
\begin{equation}
	\int_{S(-ck,\epsilon)}f(z) dz = e^{ickt} \int_{S^+(0,\epsilon)} \left(\frac{e^{-i z' t}}{z' - 2ck}\right)\frac{1}{z'}dz'
\end{equation}
$$
The expression in parentheses is analytic on a disk of radius $2ck$ centered at $z' = 0$. Since the integral takes place along a semicirucle with radius $r < 2ck$, the analyticity condition holds for the integration path so we can expand as a power series. 
$$
\begin{equation}
	u(z) = \left(a_0 + \sum_{n = 1}^{\infty}a_n z^n\right)\frac{1}{z} = \frac{a_0}{z} + \sum_{n = 1}^{\infty}a_n z^{n - 1} = \frac{a_0}{z} + P(z)
\end{equation}
$$
where we collected the power series into the function $P(z)$. Since $P(z)$ is a power series, it is bounded by some maximum value $M$ along the semicircular contour. The integral over this term is then bounded by
$$
\begin{equation}
	\left|\int_{S(0,\epsilon)}P(z)dz\right| \leq \int_{S(0,\epsilon)}M |dz| =  M \pi\epsilon.
\end{equation}
$$
Since $\lim_{\epsilon \rightarrow 0} M\pi \epsilon = 0$, the same applies to the contour integral over $P(z)$. What remains is the integral over the singular term
$$
\begin{equation}
	\int_{S(-ck,\epsilon)}f(z) dz = -\frac{e^{ickt}}{2ck}\int_{S^+(\epsilon,0)} \frac{dz}{z} = -\frac{i e^{ickt}}{2ck}\int_{-\pi}^0 d\theta = -\frac{i\pi e^{ickt}}{2ck}
\end{equation}
$$
Since the contour is a semicircle of radius $\epsilon$, we define $z = \epsilon e^{i\theta}$ so that $dz = i\epsilon e^{i\theta} d\theta$.

The integral over the other pole goes exactly the same to find
$$
\begin{equation}
	\int_{S^+(ck,\epsilon)}f(z) dz = \frac{i\pi e^{-ickt}}{2ck}
\end{equation}
$$
Putting it all together,
$$
\begin{equation}
	\Omega(\mathbf{k},t) = \frac{i\pi}{2ck}\left(e^{ickt} - e^{-ickt}\right).
\end{equation}
$$
$$
\begin{equation}
	G(\mathbf{r},t) = \frac{1}{(2\pi)^2}\frac{c}{2 \epsilon_0 r} \int_{0}^\infty \left(e^{ik(r + ct)} + e^{-ik(r + ct)} - e^{ik(r - ct)} - e^{-ik(r - ct)}\right) dk
\end{equation}
$$

$$
\begin{equation}
	G(\mathbf{r},t) = -\frac{1}{(2\pi)^2}\frac{c}{2 \epsilon_0 r} \int_{0}^\infty \left(e^{ik(r - ct)} + e^{-ik(r - ct)}\right) dk
\end{equation}
$$
$$
\begin{equation}
	G(\mathbf{r},t) = \frac{c}{4\pi \epsilon_0 r}\delta(r - ct)
\end{equation}
